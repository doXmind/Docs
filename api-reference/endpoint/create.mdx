---
title: 'Create Document'
api: 'POST /v1/documents'
description: 'Create a new document with AI-powered capabilities'
---

## Request

<ParamField body="title" type="string" required>
  The title of the document
</ParamField>

<ParamField body="type" type="string" default="document">
  The type of document. Options: `document`, `report`, `article`, `research`, `technical`
</ParamField>

<ParamField body="content" type="string">
  Initial content for the document
</ParamField>

<ParamField body="metadata" type="object">
  Additional metadata for the document
  
  <Expandable title="properties">
    <ParamField body="tags" type="array">
      Array of tags for categorization
    </ParamField>
    
    <ParamField body="department" type="string">
      Department or team name
    </ParamField>
    
    <ParamField body="project" type="string">
      Associated project identifier
    </ParamField>
    
    <ParamField body="custom" type="object">
      Custom key-value pairs
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="templateId" type="string">
  ID of a template to use for the document
</ParamField>

<ParamField body="aiSettings" type="object">
  AI configuration for the document
  
  <Expandable title="properties">
    <ParamField body="autoEnhance" type="boolean" default="true">
      Enable automatic content enhancement
    </ParamField>
    
    <ParamField body="suggestionLevel" type="string" default="moderate">
      AI suggestion aggressiveness: `minimal`, `moderate`, `aggressive`
    </ParamField>
    
    <ParamField body="language" type="string" default="en">
      Primary language for AI operations
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  The created document object

  <Expandable title="properties">
    <ResponseField name="id" type="string">
      Unique identifier for the document
    </ResponseField>

    <ResponseField name="title" type="string">
      Document title
    </ResponseField>

    <ResponseField name="type" type="string">
      Document type
    </ResponseField>

    <ResponseField name="content" type="string">
      Document content
    </ResponseField>

    <ResponseField name="status" type="string">
      Document status: `draft`, `in_progress`, `review`, `published`
    </ResponseField>

    <ResponseField name="metadata" type="object">
      Document metadata
    </ResponseField>

    <ResponseField name="aiSettings" type="object">
      AI configuration
    </ResponseField>

    <ResponseField name="wordCount" type="integer">
      Current word count
    </ResponseField>

    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of creation
    </ResponseField>

    <ResponseField name="updatedAt" type="string">
      ISO 8601 timestamp of last update
    </ResponseField>

    <ResponseField name="createdBy" type="object">
      User who created the document
      
      <Expandable title="properties">
        <ResponseField name="id" type="string">
          User ID
        </ResponseField>
        
        <ResponseField name="name" type="string">
          User name
        </ResponseField>
        
        <ResponseField name="email" type="string">
          User email
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

  }'

const response = await fetch('https://api.docmindllm.com/v1/documents', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    title: 'Q4 2024 Financial Report',
    type: 'report',
    content: 'Executive Summary\n\nThis report presents the financial performance...',
    metadata: {
      tags: ['finance', 'quarterly', '2024'],
      department: 'Finance',
      project: 'Q4-REPORT-2024'
    },
    aiSettings: {
      autoEnhance: true,
      suggestionLevel: 'moderate',
      language: 'en'
    }
  })
});

const document = await response.json();

```python Python
import requests

response = requests.post(
    'https://api.docmindllm.com/v1/documents',
    headers={
        'Authorization': 'Bearer YOUR_API_KEY',
        'Content-Type': 'application/json'
    },
    json={
        'title': 'Q4 2024 Financial Report',
        'type': 'report',
        'content': 'Executive Summary\n\nThis report presents the financial performance...',
        'metadata': {
            'tags': ['finance', 'quarterly', '2024'],
            'department': 'Finance',
            'project': 'Q4-REPORT-2024'
        },
        'aiSettings': {
            'autoEnhance': True,
            'suggestionLevel': 'moderate',
            'language': 'en'
        }
    }
)

document = response.json()
```
